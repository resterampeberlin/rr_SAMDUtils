38void SAMDGCLK::enable(uint8_t divisor) {
39 // Divide the 48MHz clock source by divisor 1 : 48MHz / 1 = 48MHz
40 // Select Generic Clock (GCLK) 0
41 GCLK->GENDIV.reg = GCLK_GENDIV_DIV(divisor) | GCLK_GENDIV_ID(id);
42 SAMDGCLKSync(STATUS.bit.SYNCBUSY);
43
44 GCLK->GENCTRL.reg = GCLK_GENCTRL_IDC | // Set the duty cycle to 50/50 HIGH/LOW
45 GCLK_GENCTRL_GENEN | // Enable GCLK0
46 GCLK_GENCTRL_SRC_DFLL48M | // Set the 48MHz clock source
47 GCLK_GENCTRL_ID(id); // Select GCLK0
48 SAMDGCLKSync(STATUS.bit.SYNCBUSY);
49}

51void SAMDGCLK::feedTCC(uint32_t tcc) {
52 uint16_t gen_gclk[] = {GCLK_CLKCTRL_GEN_GCLK0, GCLK_CLKCTRL_GEN_GCLK1, GCLK_CLKCTRL_GEN_GCLK2,
53 GCLK_CLKCTRL_GEN_GCLK3, GCLK_CLKCTRL_GEN_GCLK4, GCLK_CLKCTRL_GEN_GCLK5,
54 GCLK_CLKCTRL_GEN_GCLK6, GCLK_CLKCTRL_GEN_GCLK7};
55
56 GCLK->CLKCTRL.reg = GCLK_CLKCTRL_CLKEN | // Enable GCLKx to TCCs
57 gen_gclk[id] | // Select GCLKx
58 tcc; // Feed GCLKx to TCCs
59 SAMDGCLKSync(STATUS.bit.SYNCBUSY);
60}
